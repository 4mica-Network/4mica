## Overview

Client integration handles HTTP 402 responses from your server and manages the payment flow. The 4Mica client automatically:

- Opens payment tabs when needed
- Signs payment guarantees with the user's wallet
- Retries requests with valid payment credentials
- Manages payment state and errors

## TypeScript/JavaScript

### Fetch Wrapper

The `@x402/fetch` wrapper automatically handles 402 responses:

<Steps>
  <Step title="Install the package">
    ```bash
    npm install @x402/fetch
    ```
  </Step>

  <Step title="Replace fetch with x402Fetch">
    ```typescript

      method: 'GET',
    ```

  </Step>

  <Step title="Configure wallet connection">

  </Step>

### Manual Payment Handling

For custom implementations, handle 402 responses manually:

```typescript
import { createTab, signGuarantee } from '@x402/client';
import { useAccount } from 'wagmi';

export function usePaymentFlow() {
  const { address } = useAccount();

  const handlePayment = async (paymentRequirement) => {
    try {
      // Step 1: Open a payment tab
      const tab = await createTab({
        amount: paymentRequirement.amount,
        currency: paymentRequirement.currency,
        facilitatorId: paymentRequirement.facilitatorId,
      });

      // Step 2: Wait for user to fund tab (this opens a modal)
      await tab.waitForFunding();

      // Step 3: Sign the payment guarantee
      const guarantee = await signGuarantee({
        tabId: tab.id,
        validationPolicy: paymentRequirement.validationPolicy,
      });

      return guarantee;
    } catch (error) {
      console.error('Payment flow failed:', error);
      throw error;
    }
  };

  return { handlePayment };
}
```

## React Integration

### Hook for Payment Handling

```typescript
import { useState } from 'react';
import { useAccount } from 'wagmi';
import { x402Fetch } from '@x402/fetch';

export function useProtectedApi() {
  const { address } = useAccount();
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  const fetchProtected = async (url: string, options = {}) => {
    setLoading(true);
    setError(null);

    try {
      const response = await x402Fetch(url, {
        ...options,
        headers: {
          ...options.headers,
          'X-User-Address': address,
        },
      });

      if (!response.ok) {
        throw new Error(`API error: ${response.statusText}`);
      }

      return await response.json();
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Unknown error'));
      throw err;
    } finally {
      setLoading(false);
    }
  };

  return { fetchProtected, loading, error };
}
```

### Component Example

```typescript
import { useProtectedApi } from './useProtectedApi';
import { useEffect, useState } from 'react';

export function ProtectedContent() {
  const { fetchProtected, loading, error } = useProtectedApi();
  const [data, setData] = useState(null);

  useEffect(() => {
    (async () => {
      try {
        const result = await fetchProtected(
          'https://api.myservice.com/protected'
        );
        setData(result);
      } catch (err) {
        console.error(err);
      }
    })();
  }, []);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;
  if (!data) return <div>No data</div>;

  return <div>{JSON.stringify(data, null, 2)}</div>;
}
```

### Manual Payment Flow

```python
from mica_x402 import create_tab, sign_guarantee
import asyncio

async def handle_payment(payment_requirement):
    try:
        # Create payment tab
        tab = await create_tab(
            amount=payment_requirement['amount'],
            currency=payment_requirement['currency'],
            facilitator_id=payment_requirement['facilitatorId'],
        )

        # Wait for funding
        await tab.wait_for_funding()

        # Sign guarantee
        guarantee = await sign_guarantee(
            tab_id=tab.id,
            validation_policy=payment_requirement['validationPolicy'],
        )

        return guarantee
    except Exception as e:
        print(f'Payment flow failed: {e}')
        raise

# Use in your requests
try:
    response = requests.get('https://api.myservice.com/protected')
except requests.exceptions.HTTPError as e:
    if e.response.status_code == 402:
        guarantee = asyncio.run(
            handle_payment(e.response.json()['paymentRequired'])
        )
        # Retry with guarantee
        response = requests.get(
            'https://api.myservice.com/protected',
            headers={'X-Payment-Guarantee': guarantee.id}
        )
```

## Payment State Management

Track payment status throughout the flow:

```typescript
export function usePaymentState() {
  const [state, setState] = useState({
    status: 'idle', // idle | pending | verified | failed
    tabId: null,
    guaranteeId: null,
    amount: null,
    error: null,
  });

  const initiatePayment = async (requirement) => {
    setState((prev) => ({ ...prev, status: 'pending' }));
    
    try {
      const guarantee = await handlePaymentFlow(requirement);
      setState((prev) => ({
        ...prev,
        status: 'verified',
        guaranteeId: guarantee.id,
        amount: requirement.amount,
      }));
      return guarantee;
    } catch (error) {
      setState((prev) => ({
        ...prev,
        status: 'failed',
        error: error.message,
      }));
      throw error;
    }
  };

  return { state, initiatePayment };
}
```

## Error Handling

Handle different payment errors gracefully:

```typescript
export function usePaymentErrorHandler() {
  const handleError = (error: PaymentError) => {
    switch (error.code) {
      case 'NO_WALLET':
        return 'Please connect your wallet';
      case 'INSUFFICIENT_BALANCE':
        return 'Not enough balance to complete payment';
      case 'USER_REJECTED':
        return 'Transaction was rejected';
      case 'INVALID_GUARANTEE':
        return 'Payment guarantee is invalid';
      case 'NETWORK_ERROR':
        return 'Network error - please try again';
      default:
        return 'Payment failed - please try again';
    }
  };

  return { handleError };
}
```

## Best Practices

<Tip>
  - Always show loading states during payment processing
  - Provide clear error messages to users
  - Retry failed requests with exponential backoff
  - Store successful payment receipts for later verification
  - Handle wallet disconnection gracefully
</Tip>
