## What are Verified Payments?

Verified Payments (V2 guarantees) bind payment credentials to on-chain validation outcomes. Instead of trusting a facilitator's claim that payment was processed, the payment is only considered valid after a job validator confirms work was completed on-chain.

### V1 vs V2

| Feature | V1 Standard | V2 Verified |
|---------|-----------|-----------|
| Payment guarantee | Signed by facilitator | Signed + on-chain validated |
| Validation | Facilitator-based | ERC-8004 job validator |
| Use case | General API access | Work that requires proof |
| Settlement | Immediate | After validation |
| Complexity | Lower | Higher |

## Setting Up V2

### Step 1: Enable V2 for Your Scheme

When creating or updating your scheme, specify `guaranteeMode: 'v2'`:

<CodeGroup>
  ```typescript
  import { schemeClient } from '@4mica/x402';

  const scheme = await schemeClient.create({
    facilitatorId: facilitator.id,
    name: 'Verified Payments',
    guaranteeMode: 'v2', // Enable V2
    networks: ['eip155:8453'],
  });
  ```

  ```python
  from mica_x402 import scheme_client

  scheme = await scheme_client.create(
      facilitator_id=facilitator.id,
      name='Verified Payments',
      guarantee_mode='v2',  # Enable V2
      networks=['eip155:8453'],
  )
  ```
</CodeGroup>

### Step 2: Configure Validation Policy

Define what on-chain conditions must be met:

<CodeGroup>
  ```typescript
  import { createValidationPolicy } from '@4mica/x402';

  const policy = await createValidationPolicy({
    schemeId: scheme.id,
    validatorAddress: '0x...', // Your validator contract
    method: 'erc8004',
    requirements: {
      minimumScore: 80,
      timeLimit: 3600, // 1 hour
      allowedJobs: ['image-generation', 'text-analysis'],
    },
  });
  ```

  ```python
  from mica_x402 import create_validation_policy

  policy = await create_validation_policy(
      scheme_id=scheme.id,
      validator_address='0x...',
      method='erc8004',
      requirements={
          'minimum_score': 80,
          'time_limit': 3600,
          'allowed_jobs': ['image-generation', 'text-analysis'],
      },
  )
  ```
</CodeGroup>

## Server-Side Implementation

### Require V2 Guarantees

Update your middleware to require verified payments:

<CodeGroup>
  ```typescript
  const paymentMiddleware = await paymentMiddlewareFromConfig({
    operatorUrl: process.env.NEXT_PUBLIC_4MICA_OPERATOR_URL,
    facilitatorSecret: process.env.4MICA_FACILITATOR_SECRET,
    schemeClass: FourMicaEvmScheme,
    guaranteeMode: 'v2', // Require V2
    validationPolicy: process.env.4MICA_VALIDATION_POLICY,
  });

  app.post('/api/process-job', paymentMiddleware, async (req, res) => {
    // At this point, payment is NOT yet verified
    // Schedule the job and include the guarantee ID
    const jobId = await scheduleJob({
      ...req.body,
      guaranteeId: req.payment.guaranteeId,
    });

    res.json({ jobId });
  });
  ```

  ```python
  payment_middleware = await fastapi_payment_middleware_from_config(
      operator_url=os.getenv('NEXT_PUBLIC_4MICA_OPERATOR_URL'),
      facilitator_secret=os.getenv('4MICA_FACILITATOR_SECRET'),
      guarantee_mode='v2',
      validation_policy=os.getenv('4MICA_VALIDATION_POLICY'),
  )

  @app.post('/api/process-job')
  async def process_job(req: JobRequest, payment=Depends(payment_middleware)):
      # Schedule job with guarantee ID
      job_id = await schedule_job({
          **req.dict(),
          'guarantee_id': payment.guarantee_id,
      })
      return {'jobId': job_id}
  ```
</CodeGroup>

### Post-Job Validation

After the job completes, validate and settle the payment:

<CodeGroup>
  ```typescript
  import { validateGuarantee } from '@4mica/x402';

  // After job completes
  async function settlePayment(jobId: string, guaranteeId: string, score: number) {
    // Submit validation result
    const validation = await validateGuarantee({
      guaranteeId,
      jobId,
      score, // 0-100
      metadata: {
        completedAt: new Date(),
        workerAddress: '0x...',
      },
    });

    console.log(`Payment validated: ${validation.id}`);
    
    // Facilitate settlement
    await facilitatorClient.settlePayment({
      validationId: validation.id,
      guaranteeId,
    });
  }

  // In your job completion handler
  app.post('/api/complete-job', async (req, res) => {
    const { jobId, score } = req.body;
    const job = await getJob(jobId);

    try {
      await settlePayment(jobId, job.guaranteeId, score);
      res.json({ status: 'settled' });
    } catch (error) {
      res.status(400).json({ error: error.message });
    }
  });
  ```

  ```python
  from mica_x402 import validate_guarantee, facilitator_client

  async def settle_payment(job_id: str, guarantee_id: str, score: int):
      # Submit validation result
      validation = await validate_guarantee(
          guarantee_id=guarantee_id,
          job_id=job_id,
          score=score,
          metadata={
              'completed_at': datetime.now(),
              'worker_address': '0x...',
          },
      )

      # Settle payment
      await facilitator_client.settle_payment(
          validation_id=validation.id,
          guarantee_id=guarantee_id,
      )

  @app.post('/api/complete-job')
  async def complete_job(req: CompleteJobRequest):
      job = await get_job(req.job_id)
      
      try:
          await settle_payment(req.job_id, job.guarantee_id, req.score)
          return {'status': 'settled'}
      except Exception as e:
          return {'error': str(e)}, 400
  ```
</CodeGroup>

## Client-Side Flow

The client handles V2 verification automatically, but provides visibility:

```typescript
import { x402Fetch } from '@x402/fetch';

async function submitJob(data: JobData) {
  // Submit with automatic V2 handling
  const response = await x402Fetch('/api/process-job', {
    method: 'POST',
    body: JSON.stringify(data),
  });

  const { jobId } = await response.json();
  
  // Job is now pending validation
  console.log(`Job submitted: ${jobId}`);
  
  // You can monitor the guarantee status
  const guarantee = await fetchGuaranteeStatus(jobId);
  console.log(`Guarantee status: ${guarantee.status}`);
  
  return jobId;
}
```

## Validation Events

Monitor and react to validation events:

```typescript
// Server webhook for validation events
app.post('/webhooks/4mica', async (req, res) => {
  const event = verifyWebhook(req.body);

  if (event.type === 'validation.completed') {
    const { jobId, score, status } = event.data;
    
    if (status === 'passed') {
      // Unlock premium features or update records
      await activatePremiumFeatures(jobId);
    } else {
      // Refund or retry
      await refundPayment(event.data.guaranteeId);
    }
  }

  res.json({ received: true });
});
```

## Best Practices

<Warning>
  **Security**: Never rely solely on client-side signals for V2 validation. Always validate on the server before granting access or settling payments.
</Warning>

<Tip>
  **Scoring Guidelines:**
  - 90-100: Perfect completion, grant full features
  - 70-89: Satisfactory, grant most features
  - 50-69: Incomplete, partial refund
  - 0-49: Failed, full refund
</Tip>

## Common Patterns

### Premium Content Unlock

```typescript
// Content requires proof of validation
app.get('/api/premium/:contentId', paymentMiddleware, async (req, res) => {
  const { guaranteeId } = req.payment;
  
  // Check if V2 guarantee was validated
  const validation = await getValidation(guaranteeId);
  
  if (validation.status !== 'passed') {
    return res.status(403).json({
      error: 'Content locked until validation',
      guaranteeId,
    });
  }

  // Return premium content
  res.json({ content: getPremiumContent(contentId) });
});
```

### Tiered Access

```typescript
app.get('/api/resource', paymentMiddleware, async (req, res) => {
  const validation = await getValidation(req.payment.guaranteeId);
  
  const accessLevel = {
    passed: 'full',
    pending: 'limited',
    failed: 'none',
  }[validation.status];

  res.json({
    data: getResourceByAccessLevel(accessLevel),
    validation: { status: validation.status, score: validation.score },
  });
});
```
