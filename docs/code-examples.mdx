### Handle 402 Responses in React

```typescript
import { useState, useEffect } from 'react';
import { x402Fetch } from '@x402/fetch';
import { useAccount } from 'wagmi';

export function ProtectedContent() {
  const { address, isConnected } = useAccount();
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const fetchContent = async () => {
    if (!isConnected) {
      setError('Please connect your wallet');
      return;
    }

    setLoading(true);
    setError(null);

    try {
      const response = await x402Fetch('/api/protected', {
        method: 'GET',
        headers: {
          'X-User-Address': address,
        },
      });

      const result = await response.json();
      setData(result);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Unknown error');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="p-4">
      <button
        onClick={fetchContent}
        disabled={!isConnected || loading}
        className="px-4 py-2 bg-blue-500 text-white rounded disabled:opacity-50"
      >
        {loading ? 'Loading...' : 'Fetch Protected Content'}
      </button>

      {error && <div className="text-red-500 mt-2">{error}</div>}

      {data && (
        <div className="mt-4 p-4 bg-gray-100 rounded">
          <pre>{JSON.stringify(data, null, 2)}</pre>
        </div>
      )}
    </div>
  );
}
```

### Custom Payment Handler

<CodeGroup>
  ```typescript
  import { createTab, signGuarantee } from '@x402/client';
  import { useAccount, useSignMessage } from 'wagmi';

  export async function handleCustomPayment(requirement) {
    // Step 1: Create a payment tab
    const tab = await createTab({
      amount: requirement.amount,
      currency: requirement.currency,
      facilitatorId: requirement.facilitatorId,
      metadata: {
        requestPath: requirement.requestPath,
        userId: requirement.userId,
      },
    });

    console.log(`Created tab: ${tab.id}`);

    // Step 2: Wait for funding in the tab
    await tab.waitForFunding({
      timeout: 5 * 60 * 1000, // 5 minutes
      pollInterval: 1000,
    });

    console.log(`Tab funded with ${tab.balance}`);

    // Step 3: Sign a payment guarantee
    const message = JSON.stringify({
      tabId: tab.id,
      amount: requirement.amount,
      timestamp: Date.now(),
      nonce: Math.random(),
    });

    const { data: signature } = await signMessage({ message });

    const guarantee = {
      tabId: tab.id,
      amount: requirement.amount,
      signature,
      validationPolicy: requirement.validationPolicy,
    };

    console.log(`Created guarantee: ${guarantee.signature}`);

    return guarantee;
  }
  ```

### Webhook Validation

```typescript
import { verifyWebhook } from '@4mica/x402';
import express from 'express';

app.post('/webhooks/4mica', express.json(), (req, res) => {
  try {
    // Verify the webhook signature
    const event = verifyWebhook(
      req.body,
      req.headers['x-4mica-signature'] as string,
      process.env.4MICA_WEBHOOK_SECRET!
    );

    // Handle different event types
    switch (event.type) {
      case 'payment.verified': {
        const { guaranteeId, amount, tabId } = event.data;
        console.log(`Payment verified: ${amount} ${event.data.currency}`);
        // Update your database
        break;
      }

      case 'tab.opened': {
        const { tabId, user } = event.data;
        console.log(`Tab opened by ${user}`);
        // Track user behavior
        break;
      }

      case 'settlement.completed': {
        const { settlementId, amount } = event.data;
        console.log(`Settlement completed: ${amount}`);
        // Record settlement in accounting system
        break;
      }

      case 'payment.failed': {
        const { guaranteeId, reason } = event.data;
        console.log(`Payment failed: ${reason}`);
        // Handle failure
        break;
      }
    }

    res.json({ received: true });
  } catch (error) {
    console.error('Webhook validation failed:', error);
    res.status(401).json({ error: 'Invalid signature' });
  }
});
```

### Batch Settlement

```typescript
import { facilitatorClient } from '@4mica/x402';

async function dailySettlement() {
  // Get all verified guarantees from the last 24 hours
  const guarantees = await facilitatorClient.getGuarantees({
    status: 'verified',
    createdAfter: new Date(Date.now() - 24 * 60 * 60 * 1000),
  });

  console.log(`Found ${guarantees.length} guarantees to settle`);

  if (guarantees.length === 0) {
    console.log('No guarantees to settle');
    return;
  }

  // Group by scheme for efficient settlement
  const byScheme = guarantees.reduce((acc, g) => {
    const key = g.schemeId;
    if (!acc[key]) acc[key] = [];
    acc[key].push(g);
    return acc;
  }, {} as Record<string, any[]>);

  // Process each scheme
  for (const [schemeId, schemeGuarantees] of Object.entries(byScheme)) {
    const totalAmount = schemeGuarantees.reduce(
      (sum, g) => sum + parseFloat(g.amount),
      0
    );

    console.log(
      `Settling ${schemeGuarantees.length} guarantees for scheme ${schemeId}: ${totalAmount} ETH`
    );

    try {
      const settlement = await facilitatorClient.settle({
        guaranteeIds: schemeGuarantees.map((g) => g.id),
        metadata: {
          period: new Date().toISOString().split('T')[0],
          reason: 'daily_settlement',
        },
      });

      console.log(`Settlement ${settlement.id} initiated`);
    } catch (error) {
      console.error(`Settlement failed for scheme ${schemeId}:`, error);
    }
  }
}

// Run daily
setInterval(dailySettlement, 24 * 60 * 60 * 1000);
```

### Monitor Payment Status

```typescript
export function usePaymentMonitor(guaranteeId: string) {
  const [status, setStatus] = useState('pending');
  const [details, setDetails] = useState(null);

  useEffect(() => {
    const pollInterval = setInterval(async () => {
      try {
        const response = await fetch(
          `/api/guarantees/${guaranteeId}`,
          {
            headers: {
              'Authorization': `Bearer ${process.env.NEXT_PUBLIC_API_KEY}`,
            },
          }
        );

        const guarantee = await response.json();
        setStatus(guarantee.status);
        setDetails(guarantee);

        // Stop polling once settled
        if (guarantee.status === 'settled') {
          clearInterval(pollInterval);
        }
      } catch (error) {
        console.error('Failed to fetch guarantee status:', error);
      }
    }, 5000); // Poll every 5 seconds

    return () => clearInterval(pollInterval);
  }, [guaranteeId]);

  return { status, details };
}

// Usage in React
export function PaymentStatus({ guaranteeId }) {
  const { status, details } = usePaymentMonitor(guaranteeId);

  const statusMessages = {
    pending: 'Payment processing...',
    verified: 'Payment verified, waiting for validation...',
    settled: '✓ Payment settled successfully',
    failed: '✗ Payment failed',
  };

  return (
    <div className="p-4">
      <p>{statusMessages[status]}</p>
      {details && (
        <div className="mt-2 text-sm text-gray-600">
          Amount: {details.amount} {details.currency}
        </div>
      )}
    </div>
  );
}
```

### Error Recovery

```typescript
export async function fetchWithRetry(
  url: string,
  options: RequestInit = {},
  maxRetries: number = 3
): Promise<Response> {
  let lastError: Error | null = null;

  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      const response = await x402Fetch(url, options);

      if (response.ok) {
        return response;
      }

      if (response.status === 402) {
        // Payment required - this is expected, not a retry case
        return response;
      }

      if (response.status >= 500) {
        // Server error - retry
        throw new Error(`Server error: ${response.statusText}`);
      }

      // 4xx client error - don't retry
      return response;
    } catch (error) {
      lastError = error instanceof Error ? error : new Error(String(error));

      if (attempt < maxRetries) {
        // Exponential backoff
        const delay = Math.pow(2, attempt - 1) * 1000;
        console.log(
          `Attempt ${attempt} failed, retrying in ${delay}ms...`
        );
        await new Promise((resolve) => setTimeout(resolve, delay));
      }
    }
  }

  throw lastError || new Error('Failed after all retries');
}

// Usage
try {
  const response = await fetchWithRetry('/api/protected');
  const data = await response.json();
  console.log(data);
} catch (error) {
  console.error('Request failed:', error);
}
```
